extract_mime_type (io)

If a Shrine::UploadedFile was given, it returns its MIME type, since that value was already determined by this analyzer. Otherwise it calls a built-in analyzer or a custom one.

def extract_mime_type(io)
  analyzer = opts[:mime_type_analyzer]

  if io.respond_to?(:mime_type)
    io.mime_type
  elsif analyzer.is_a?(Symbol)
    send(:"_extract_mime_type_with_#{analyzer}", io)
  else
    analyzer.call(io)
  end
end
