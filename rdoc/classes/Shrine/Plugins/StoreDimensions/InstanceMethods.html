extract_dimensions (io)

If the io is an uploaded file, copies its dimensions, otherwise calls the predefined or custom analyzer.

[show source] 
# File lib/shrine/plugins/store_dimensions.rb, line 50
def extract_dimensions(io)
  analyzer = opts[:dimensions_analyzer]

  dimensions = if io.respond_to?(:width) && io.respond_to?(:height)
    [io.width, io.height]
  elsif analyzer.is_a?(Symbol)
    send(:"_extract_dimensions_with_#{analyzer}", io)
  else
    analyzer.call(io)
  end

  io.rewind

  dimensions
end

def extract_metadata(io, context)
  width, height = extract_dimensions(io)

  super.update(
    "width"  => width,
    "height" => height,
  )
end
