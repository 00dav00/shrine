def _promote
  if background_promote = shrine_class.opts[:background_promote]
    data = {
      "uploaded_file" => get.to_json,
      "record"        => [record.class.to_s, record.id],
      "attachment"    => name,
    }

    instance_exec(data, &background_promote) if promote?(get)
  else
    super
  end
end
