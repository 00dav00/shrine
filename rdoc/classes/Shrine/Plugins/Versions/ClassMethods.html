def uploaded_file(object, &block)
  if (hash = object).is_a?(Hash) && !hash.key?("storage")
    hash.inject({}) do |result, (name, data)|
      next result if !version?(name)
      result.update(name.to_sym => uploaded_file(data, &block))
    end
  else
    super
  end
end
